using System;
using System.Windows.Forms;

namespace MDI_ImageViewer
{
    public partial class MDIParent : Form
    {
        public MDIParent()
        {
            InitializeComponent();
        }

        private void MDIParent_Load(object sender, EventArgs e)
        {
            this.IsMdiContainer = true; // Make it MDI container
        }

        private void openToolStripMenuItem_Click(object sender, EventArgs e)
        {
            OpenFileDialog ofd = new OpenFileDialog();
            ofd.Filter = "Image Files|*.jpg;*.jpeg;*.png;*.bmp;*.gif";

            if (ofd.ShowDialog() == DialogResult.OK)
            {
                ChildForm child = new ChildForm(ofd.FileName);
                child.MdiParent = this;
                child.Show();
            }
        }

        private void closeToolStripMenuItem_Click(object sender, EventArgs e)
        {
            if (this.ActiveMdiChild != null)
                this.ActiveMdiChild.Close();
        }

        private void exitToolStripMenuItem_Click(object sender, EventArgs e)
        {
            this.Close();
        }
    }
}

Child Form

csharp
using System;
using System.Drawing;
using System.Windows.Forms;

namespace MDI_ImageViewer
{
    public partial class ChildForm : Form
    {
        private Image originalImage;
        private float zoomFactor = 1.0f;

        public ChildForm(string filePath)
        {
            InitializeComponent();
            originalImage = Image.FromFile(filePath);
            pictureBox1.Image = originalImage;
            pictureBox1.SizeMode = PictureBoxSizeMode.Zoom;
        }

        private void btnZoomIn_Click(object sender, EventArgs e)
        {
            zoomFactor += 0.1f;
            ApplyZoom();
        }

        private void btnZoomOut_Click(object sender, EventArgs e)
        {
            if (zoomFactor > 0.2f)
            {
                zoomFactor -= 0.1f;
                ApplyZoom();
            }
        }

        private void btnReset_Click(object sender, EventArgs e)
        {
            zoomFactor = 1.0f;
            ApplyZoom();
        }

        private void ApplyZoom()
        {
            if (pictureBox1.Image != null)
            {
                pictureBox1.Width = (int)(originalImage.Width * zoomFactor);
                pictureBox1.Height = (int)(originalImage.Height * zoomFactor);
            }
        }
    }
}

The Reset button is used to bring the image back to its original size (the size it was loaded with), no matter how much the user zooms in or zooms out.

Example:

  • Suppose your image is 800 × 600 pixels.

  • When you Zoom In, we increase the zoomFactor (e.g., 1.2, 1.5 …), so the PictureBox becomes bigger than 800 × 600.

  • When you Zoom Out, we decrease the zoomFactor (e.g., 0.8, 0.6 …), so the PictureBox becomes smaller.

  • When you click Reset, we set:

    csharp
    zoomFactor = 1.0f;
pictureBox1.Width = originalImage.Width;
pictureBox1.Height = originalImage.Height;

    → which restores the original size of the image (same as when it was first opened).

👉 In short:

  • Zoom In → Makes image bigger

  • Zoom Out → Makes image smaller

  • Reset → Back to original size

1️⃣ Common mistake

Many beginners do this in ApplyZoom():

csharp

  • This resets the reference but the PictureBox size remains scaled from previous zooms.

  • Reset only works if you restore the PictureBox size, not just the image.

2️⃣ Correct Reset logic

csharp
private void btnReset_Click(object sender, EventArgs e)
{
    zoomFactor = 1.0f; // reset zoom factor
    ApplyZoom();        // apply the original size
}

private void ApplyZoom()
{
    if (originalImage != null)
    {
        pictureBox1.Width = (int)(originalImage.Width * zoomFactor);
        pictureBox1.Height = (int)(originalImage.Height * zoomFactor);
    }
}

Important:

  • pictureBox1.SizeMode should remain PictureBoxSizeMode.Zoom or Normal depending on your layout.

  • If you have Dock = Fill, the Width/Height changes might not be visible—so remove Dock = Fill and put PictureBox inside a scrollable Panel.

✅ Extra tip

  • Wrap the PictureBox inside a Panel with AutoScroll = true.

  • This allows zooming without losing image or going out of bounds.

  • Then Reset works perfectly because you’re just resizing the PictureBox, not the Panel.
